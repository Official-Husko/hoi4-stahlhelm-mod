	STH_helmet_equipment = {
		year = 1916

		is_archetype = yes
		picture = STH_archetype_helmet_equipment
		is_buildable = no
		# "armor" added alongside the real types, not in place of them -
		# confirmed via the "Infantry Equipment Designer" mod's own
		# train_equipment archetype ("type = { armor train }"): the
		# Equipment Designer screen only recognizes "armor" as a marker
		# that an archetype is designable, while infantry/support stay
		# intact for actual unit "need" matching.
		type = {
			armor
			infantry
			support
		}
		group_by = archetype

		interface_category = interface_category_land
		priority = 2000

		# No "upgrades = { }" block here on purpose - that's the OLD linear
		# upgrade-chain system, mutually exclusive with module_slots (the
		# Equipment Designer system). Neither the vanilla tank/ship/plane
		# archetypes nor the "Infantry Equipment Designer" mod's
		# infantry_equipment archetype define "upgrades" at all - having it
		# present, even empty, may be exactly what was suppressing the
		# Design button despite module_slots being defined.
		#
		# One harmless optional slot (not zero) - every known working
		# example (vanilla tanks/ships, the IED mod's own infantry_equipment)
		# has at least one real slot; testing whether a literally empty
		# module_slots block was itself the remaining blocker. Real upgrade
		# modules are still a deferred follow-up.
		module_slots = {
			liner_slot = {
				required = no
				allowed_module_categories = {
					STH_helmet_liner_type
				}
			}
		}

		active = yes

		#Misc Abilities
		reliability = 0.85
		maximum_speed = 4

		#Defensive Abilities
		defense = 2
		breakthrough = 0
		hardness = 0
		armor_value = 0

		#Space taken in convoy
		lend_lease_cost = 1

		build_cost_ic = 0.1
		resources = {
			steel = 1
		}

	}
