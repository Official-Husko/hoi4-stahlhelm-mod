# Daily 5% chance per variant that the day's whole output run comes out
# Beaded. on_daily already runs once per country with ROOT scoped to that
# country, so no every_country wrap is needed.
#
# HOI4 has no effect that reads an exact stockpile/production number into a
# variable - only boolean checks like has_equipment (>= amount). So "the
# day's output" is approximated as a fixed batch (FULL_DAY_BATCH) rather
# than a true delta. remove_equipment_from_stockpile and
# add_equipment_to_stockpile both use that same fixed amount, so the total
# equipment count is always conserved - nothing is created or destroyed,
# just relabeled.

on_actions = {
	on_daily = {
		effect = {
			if = {
				limit = { has_equipment = { helmet_equipment_m35 = 20 } }
				random_list = {
					5 = {
						remove_equipment_from_stockpile = { type = helmet_equipment_m35 amount = 20 }
						add_equipment_to_stockpile = { type = helmet_equipment_m35_beaded amount = 20 producer = ROOT }
					}
					95 = {}
				}
			}
			if = {
				limit = { has_equipment = { helmet_equipment_m40 = 20 } }
				random_list = {
					5 = {
						remove_equipment_from_stockpile = { type = helmet_equipment_m40 amount = 20 }
						add_equipment_to_stockpile = { type = helmet_equipment_m40_beaded amount = 20 producer = ROOT }
					}
					95 = {}
				}
			}
			if = {
				limit = { has_equipment = { helmet_equipment_m42 = 20 } }
				random_list = {
					5 = {
						remove_equipment_from_stockpile = { type = helmet_equipment_m42 amount = 20 }
						add_equipment_to_stockpile = { type = helmet_equipment_m42_beaded amount = 20 producer = ROOT }
					}
					95 = {}
				}
			}
		}
	}
}
