# Daily 5% chance per variant that the day's whole output run comes out
# defective (Beaded for M35/M40/M42, Ventless for M18). on_daily already
# runs once per country with ROOT scoped to that country, so no
# every_country wrap is needed.
#
# HOI4 has no effect that reads an exact stockpile/production number into a
# variable - only boolean checks like has_equipment (>= amount). So "the
# day's output" is approximated as a fixed batch (FULL_DAY_BATCH) rather
# than a true delta. There is no "remove_equipment_from_stockpile" effect in
# the engine - removal is just add_equipment_to_stockpile with a negative
# amount. Both the negative (removal) and positive (addition) calls use the
# same fixed amount, so the total equipment count is always conserved -
# nothing is created or destroyed, just relabeled.

on_actions = {
	on_daily = {
		effect = {
			if = {
				limit = { has_equipment = { STH_helmet_equipment_m35 = 20 } }
				random_list = {
					5 = {
						add_equipment_to_stockpile = { type = STH_helmet_equipment_m35 amount = -20 }
						add_equipment_to_stockpile = { type = STH_helmet_equipment_m35_beaded amount = 20 producer = ROOT }
					}
					95 = {}
				}
			}
			if = {
				limit = { has_equipment = { STH_helmet_equipment_m40 = 20 } }
				random_list = {
					5 = {
						add_equipment_to_stockpile = { type = STH_helmet_equipment_m40 amount = -20 }
						add_equipment_to_stockpile = { type = STH_helmet_equipment_m40_beaded amount = 20 producer = ROOT }
					}
					95 = {}
				}
			}
			if = {
				limit = { has_equipment = { STH_helmet_equipment_m42 = 20 } }
				random_list = {
					5 = {
						add_equipment_to_stockpile = { type = STH_helmet_equipment_m42 amount = -20 }
						add_equipment_to_stockpile = { type = STH_helmet_equipment_m42_beaded amount = 20 producer = ROOT }
					}
					95 = {}
				}
			}
			if = {
				limit = { has_equipment = { STH_helmet_equipment_m18 = 20 } }
				random_list = {
					5 = {
						add_equipment_to_stockpile = { type = STH_helmet_equipment_m18 amount = -20 }
						add_equipment_to_stockpile = { type = STH_helmet_equipment_m18_ventless amount = 20 producer = ROOT }
					}
					95 = {}
				}
			}
		}
	}
}
